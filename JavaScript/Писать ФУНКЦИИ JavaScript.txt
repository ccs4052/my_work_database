function calc(num1,num2,action){
            if(action == undefined){ //undefined (без кавычек) если action = default (тоесть пустой ничего не передаем)
                action = 'plus'; //то выполняем действие +
            }
            var res = 0; //создали переменную куда засунем ответ (локальнцю var)
            if(action =='plus'){
                res = num1 + num2;
            } else if(action == 'minus') {
                res = num1 - num2;
            }
            return res;
        }
        x = calc(1,5);
        alert('Результат: ' + x);
        x = calc(6,5,'plus');
        alert('Результат: ' + x);

function test(){

    }else{
        x.style.display ='block'; //поскольку мы его сделали скрытым если по нему еще раз нажать он сделаеться видимым

}
