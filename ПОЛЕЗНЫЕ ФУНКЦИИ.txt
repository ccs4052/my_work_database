------ array_map - прогнать весь массив и обработать какими-то действиями
			function trimALL($el) {     //функция которая прогоняет весь массив через функцию TRIM
				if(!is_array($el)){	    //если не массив тогда обработаем
					$el = trim($el);    //$el - просто элемент
					}
					return $el;
			}
			
			$array = array_map('trimALL',$array); - прогоняет весь массив через функцию TRIM
		
		
------ function __autoload ($class) {            //используеться 2 подчеркивания
			include './libs/class_'.$class.'.php';
			}
		   //а файле в котором у нас существуют все функции для работы с сайтом если написать функцию
			autoload и в ней указать имя класса который мы вызовим у нас в этом коде - то инклюд
			внутри этой функции запустит нужный файл с именем класса который мы вызвали по коду 
			в котором вызван файл с нашими функциями и в котором описана функция autoload
