

						------------------------Работа в терминале UBUNTU
						

	Запуск теминала - ctrl+alt+t     - помоему
	
	echo - выводит информацию на экран
	
	ls - показывает какие файлы и папки находяться в папке в которой мы щас находимся
	(ls -a) - отображает все файлы + те файлы названые с . перед названием
	
	mkdir - создает директорию ( mkdir test) (что-бы в зайти в папку которую мы создали cd test/)
	
	cd - команда для доступа в директорию (папку в которую нам нужно) - если не указать папку в которую мы хотим попасть cd вернет нас обратно в домашний каталог
		cd .. - вернет нас в предыдущий каталог

	pwd - покажет путь к папке в которой мы щас находимся 
	
	touch - создает файл (если файла не существует то создает) (touch текст)-создали файл текст
	
	Что-бы записать что-либо в файл который мы создали touch
		echo >текст(имя файла который мы создали) Привет мир! (привет мир запишиться в текстовій файл текст)
	
	cat - отобразит содержимое файла
	Что-бы отобразить то что записано в файле нужно 
		cat текст -Отобразит то что записано в файле текст
	
	cp - копируем файл в новый файл
		cp текст(имя файла который копируем) текст2(новая копия файла)
	
	
	ЧТО-бы посмотреть ПОМОЩЬ о команде в терминале UBUNTU
		ls(любая команда) --help   - и будет выведена справка о команде  (будет написано как ей работать)
	
	man - выводит подробную справку на команду в том числе на себя (man man) или (man ls)
	
	ЕСЛИ 2 РАЗА НАДАТЬ TAB  то покажет весь список что есть в папке 
	ЕСЛИ 1 РАЗ нажать таб во время написания например имена фйла если такой файл есть то он сам дозапишеться в строке (например первые 3 буквы написали если нажмем таб 1 раз если такое слово есть его выведет на экран)
	
	rm - удаляет файл (НО ПАПКУ УДАЛИТЬ НЕЛЬЗЯ)
		rm текст - и файл текст будет удален
		
	rmdir - удаляет папку (НО ТОЛЬКО ПУСТУЮ ПАПКУ с файлами не удалит)
	
	(rm -rf) -(КОМАНДА ОЧЕНЬ ОПАСНАЯ ЕЙ ЛУЧЬШЕ НЕ ПОЛЬЗОВАТЬСЯ можно не востановить потом файл который был ей удален) удалит папку вместе с файлами (rm -rf test)
	
	find - находит папку или файл по эго имени
		(find -name test) - test файл который ищем - покажет путь к этому файлу если он есть
		(find -name te*) - находит все папки которые начинаються на (te)
		(find -name *txt) - найдет все файлы заканчивающие на txt
		
	clear - очищает терминал от ненужной информации
	
	(grep -r текст *) - ищет файл у котого внутри записано текст 
	
	(apt-get) - нужна для того что-бы устанавливать и удалять разные программы
		(apt-get --help) - справка о команде
	УСТАНОВКА	(apt-get install test) - УСТАНАВЛИВАЕМ ФАЙЛ test
		если нужны права на установку ТО делаем такое
		(sudo apt-get install test) - Далие запросит пароль пользователя - если пароль верный то пойдет установка
	УДАЛЕНИЕ (sudo apt-get remove test) - test название программы которую удаляем	
	ЗАПУСК ПРОГРАММЫ если просто напишем название программы он ОТКРОЕТЬСЯ - ОСТАНОВКА ctrl+c и программа закроеться
	КОПИРОВАНИЕ ИНФОРМАЦИИ ИЗ ТЕРМИНАЛА ctrl+shft+c и потом echo ctrl+shift+v - ВСТАВИТЬ
