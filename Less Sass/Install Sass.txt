3.1) style.scss - пишем стили которые нада и подключаем еще разные файлы Sass например
@import "header.scss";

3) Создаем в корневой папке файл gulpfile.js


в нем пишем код 

var gulp = require('gulp');
var sass = require('gulp-sass');


gulp.task('styles', function() {
    gulp.src('scss/style.scss')
        .pipe(sass().on('error', sass.logError))
        .pipe(gulp.dest('./css/'));
});
gulp.task("styles:watch", function() {
    gulp.watch('scss/**/*.scss', ["styles"]);
});




4) Далее открываем терминал и запускаем компилятор Sass файлов

gulp style:watch
